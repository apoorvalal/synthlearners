252    def _fit_synthetic_control(self, Y: np.ndarray, N0: int, T0: int) -> PenguinResults:
253        """Fit regularized synthetic control."""
254        # Prepare data: control units (pre-treatment) as features, treated unit (pre-treatment) as target
255        Y_control_pre = Y[:N0, :T0]  # N0 x T0
256        Y_treated_pre = Y[N0:, :T0].mean(axis=0)  # T0 x 1 (average if multiple treated)
257
258        # Use adelie_synth for the core optimization
259        result = adelie_synth(
260            X=Y_control_pre.T,  # T0 x N0 (adelie expects T x N format)
261            y=Y_treated_pre,  # T0 x 1
262            l1_ratio=self.l1_ratio,
263            intercept=self.intercept,
264            noisy=self.noisy,
265        )
266
267        # Extract results
268        weights = result["weights"].flatten()
269        intercept = result["intercept"] if self.intercept else None
270        lambda_reg = result["lambda"]
271        fitter = result["fitter"]
272
273        # Generate synthetic outcomes for full time series
274        if self.intercept and intercept is not None:
275            synthetic_outcome = Y[:N0, :].T @ weights + intercept
276        else:
277            synthetic_outcome = Y[:N0, :].T @ weights
278
279        # Get treated outcome (average if multiple treated units)
280        treated_outcome = Y[N0:, :].mean(axis=0)
281
282        # Calculate treatment effects
283        treatment_effect = treated_outcome - synthetic_outcome
284        att = treatment_effect[T0:].mean()  # Average post-treatment effect
285
286        # Pre-treatment fit quality
287        pre_treatment_rmse = np.sqrt(np.mean(treatment_effect[:T0] ** 2))
288
289        return PenguinResults(
290            weights=weights,
291            intercept=intercept,
292            lambda_reg=lambda_reg,
293            treated_outcome=treated_outcome,
294            synthetic_outcome=synthetic_outcome,
295            treatment_effect=treatment_effect,
296            att=att,
297            pre_treatment_rmse=pre_treatment_rmse,
298            method=self.method,
299            l1_ratio=self.l1_ratio,
300            fitter=fitter,
301        )

303    def _fit_synthetic_did(self, Y: np.ndarray, N0: int, T0: int) -> PenguinResults:
304        """Fit regularized synthetic difference-in-differences.
305
306        Implements true SDID with two separate regularized regressions:
307        1. Vertical (unit weights): regress treated pre-treatment on control pre-treatment
308        2. Horizontal (time weights): regress control post-treatment on control pre-treatment
309        """
310        # Split data
311        Y_control_pre = Y[:N0, :T0]  # N0 x T0 - control units, pre-treatment
312        Y_control_post = Y[:N0, T0:]  # N0 x T_post - control units, post-treatment
313        Y_treated_pre = Y[N0:, :T0]  # N1 x T0 - treated units, pre-treatment
314
315        N1 = Y.shape[0] - N0
316        T_post = Y.shape[1] - T0
317
318        # REGRESSION 1: VERTICAL - Unit weights (ω)
319        # Regress treated unit pre-treatment time series on control units pre-treatment
320        Y_treated_pre_mean = Y_treated_pre.mean(
321            axis=0
322        )  # T0 x 1 (average if multiple treated)
323
324        unit_result = adelie_synth(
325            X=Y_control_pre.T,  # T0 x N0 (time x control_units)
326            y=Y_treated_pre_mean,  # T0 x 1 (treated pre-treatment trajectory)
327            l1_ratio=self.l1_ratio,
328            intercept=self.intercept,
329            noisy=self.noisy,
330        )
331        unit_weights = unit_result["weights"].flatten()  # N0 x 1
332
333        # REGRESSION 2: HORIZONTAL - Time weights (λ)
334        # Regress each control unit's post-treatment average on its pre-treatment trajectory
335        Y_control_post_unit_means = Y_control_post.mean(
336            axis=1
337        )  # N0 x 1 (each control's post-treatment mean)
338
339        time_result = adelie_synth(
340            X=Y_control_pre,  # N0 x T0 (each row is a control unit's pre-treatment trajectory)
341            y=Y_control_post_unit_means,  # N0 x 1 (each control's post-treatment average)
342            l1_ratio=self.l1_ratio,
343            intercept=self.intercept,
344            noisy=self.noisy,
345        )
346        time_weights = time_result["weights"].flatten()  # T0 x 1
347
348        # Construct SDID weight vectors following the matrix formulation
349        # Unit weights: [-ω, 1/N1, ..., 1/N1] (length N0 + N1)
350        unit_weight_vec = np.concatenate([-unit_weights, np.ones(N1) / N1])
351
352        # Time weights: [-λ, 1/T_post, ..., 1/T_post] (length T0 + T_post)
353        time_weight_vec = np.concatenate([-time_weights, np.ones(T_post) / T_post])
354
355        # Compute SDID estimate: τ̂ = ω_unit^T Y λ_time
356        sdid_estimate = unit_weight_vec.T @ Y @ time_weight_vec
357
358        # For visualization purposes, create synthetic outcomes using unit weights
359        # This shows the synthetic control trajectory (useful for plotting)
360        treated_outcome = Y[N0:, :].mean(axis=0)  # Average treated trajectory
361
362        if self.intercept and unit_result["intercept"] is not None:
363            synthetic_outcome = Y[:N0, :].T @ unit_weights + unit_result["intercept"]
364        else:
365            synthetic_outcome = Y[:N0, :].T @ unit_weights
366
367        # Treatment effect trajectory (for plotting)
368        treatment_effect = treated_outcome - synthetic_outcome
369
370        # Pre-treatment fit quality (based on unit weights regression)
371        pre_treatment_rmse = np.sqrt(np.mean(treatment_effect[:T0] ** 2))
372
373        return PenguinResults(
374            weights=unit_weights,
375            intercept=unit_result["intercept"] if self.intercept else None,
376            lambda_reg=unit_result[
377                "lambda"
378            ],  # Could also include time_result["lambda"]
379            treated_outcome=treated_outcome,
380            synthetic_outcome=synthetic_outcome,
381            treatment_effect=treatment_effect,
382            att=float(sdid_estimate),  # This is the key SDID estimate
383            pre_treatment_rmse=pre_treatment_rmse,
384            method=self.method,
385            l1_ratio=self.l1_ratio,
386            fitter=unit_result["fitter"],  # Could create combined fitter
387        )

428    ) -> np.ndarray:

434                i if i < leave_out_idx else i - 1

461        if len(treated_units) <= 1:

495    ) -> float:
496        """Compute permutation test p-value."""
497        # Get true effect
498        true_results = self.fit(

500        )
501        true_effect = np.abs(true_results.post_treatment_effect)

505        if (n - 1) <= 20:
506            print(

510        # Get control units and compute placebo effects
511        control_units = np.setdiff1d(range(Y.shape[0]), treated_units)
512        with tqdm_joblib(
513            tqdm(total=len(control_units), desc="Computing permutation test")
514        ):
515            placebo_effects = Parallel(n_jobs=self.n_jobs)(
516                delayed(self._compute_placebo_effect)(
517                    Y, control_unit, treated_units, T_pre
518                )
519                for control_unit in control_units
520            )
521
